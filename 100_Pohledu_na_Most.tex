\startluacode
    -- Načtení všech souborů v adresáři vyhovující '201*.tex'
    require "lfs"

    chapters = {}   -- new array
    for file in lfs.dir(".") do
        if file:match("201.*%.tex") then
            table.insert(chapters, file)
         end
    end

    table.sort(chapters)

    for i = 1, #chapters do
        print("Including chapter: " ..  chapters[i])
        tex.print("\\input " .. chapters[i])
    end
\stopluacode
